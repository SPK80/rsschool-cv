function countOfOnBites(number) {
  return [...number.toString(2)].filter(ch => ch === '1').length;
}

function sortByBit(arr) {
  return arr.sort((n1, n2) => {
    const d = countOfOnBites(n1) - countOfOnBites(n2);
    return d === 0
      ? n1 - n2
      : d
  })
}
